## ARIMA / Prophet

### ARIMA with statsmodels

```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.arima.model import ARIMA
from statsmodels.tsa.stattools import adfuller
from sklearn.metrics import mean_absolute_percentage_error

# Load time series
df = pd.read_csv('sales.csv', parse_dates=['date'], index_col='date')
ts = df['sales'].resample('D').sum()

# Check stationarity
result = adfuller(ts)
print(f"ADF Statistic: {result[0]:.4f}")
print(f"P-value: {result[1]:.4f}")
# If p > 0.05, need to differentiate

# Fit ARIMA
model = ARIMA(ts, order=(1, 1, 1))
fitted = model.fit()

# Forecast next 30 days
forecast = fitted.forecast(steps=30)

# Plot
plt.figure(figsize=(12, 6))
plt.plot(ts, label='Historical')
plt.plot(forecast, label='Forecast', color='red')
plt.legend()
plt.show()

# Evaluate
actual = ts[-30:]
predicted = forecast[:len(actual)]
mape = mean_absolute_percentage_error(actual, predicted)
print(f"MAPE: {mape:.4f}")
```

### Prophet (Facebook)

```python
from prophet import Prophet

# Prophet expects columns 'ds' (date) and 'y' (value)
df_prophet = pd.DataFrame({
    'ds': df.index,
    'y': df['sales']
})

# Train
model = Prophet(
    yearly_seasonality=True,
    weekly_seasonality=True,
    daily_seasonality=True,
    seasonality_mode='multiplicative'
)
model.fit(df_prophet)

# Forecast future
future = model.make_future_dataframe(periods=30)
forecast = model.predict(future)

# Plot
fig = model.plot(forecast)
fig2 = model.plot_components(forecast)
```

## LSTM for Time Series

```python
import torch
import torch.nn as nn
import numpy as np
from sklearn.preprocessing import MinMaxScaler

# Normalize data
scaler = MinMaxScaler()
ts_scaled = scaler.fit_transform(ts.values.reshape(-1, 1)).flatten()

# Create sequences
def create_sequences(data, seq_length):
    X, y = [], []
    for i in range(len(data) - seq_length):
        X.append(data[i:i+seq_length])
        y.append(data[i+seq_length])
    return np.array(X), np.array(y)

seq_length = 30
X, y = create_sequences(ts_scaled, seq_length)
X = X.reshape(-1, seq_length, 1)

# LSTM Model
class TimeSeriesLSTM(nn.Module):
    def __init__(self, hidden_size=50):
        super().__init__()
        self.lstm = nn.LSTM(input_size=1, hidden_size=hidden_size,
                            num_layers=2, batch_first=True)
        self.fc = nn.Linear(hidden_size, 1)

    def forward(self, x):
        out, _ = self.lstm(x)
        return self.fc(out[:, -1, :])

model = TimeSeriesLSTM()
criterion = nn.MSELoss()
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)

# Train
X_tensor = torch.FloatTensor(X)
y_tensor = torch.FloatTensor(y).reshape(-1, 1)

for epoch in range(100):
    optimizer.zero_grad()
    predictions = model(X_tensor)
    loss = criterion(predictions, y_tensor)
    loss.backward()
    optimizer.step()

# Predict
with torch.no_grad():
    predictions = model(X_tensor)

# Inverse transform
predictions = scaler.inverse_transform(predictions.numpy())
```

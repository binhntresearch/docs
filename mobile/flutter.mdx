```dart
class TaskCard extends StatelessWidget {

  final String status;

  final VoidCallback onDelete;

          icon: const Icon(Icons.delete, color: Colors.red),
          onPressed: onDelete,
        ),

    );
  }
}
```

## State Management — Riverpod

```dart
// providers/task_provider.dart
import 'package:riverpod/riverpod.dart';

final taskListProvider = StateNotifierProvider<TaskListNotifier, List<Task>>((ref) {
  return TaskListNotifier();
});

class TaskListNotifier extends StateNotifier<List<Task>> {
  TaskListNotifier() : super([]);

  void add(Task task) {
    state = [...state, task];
  }

  void remove(String id) {
    state = state.where((t) => t.id != id).toList();
  }

  void toggleStatus(String id) {
    state = state.map((t) {
      if (t.id == id) {
        return t.copyWith(
          status: t.status == 'done' ? 'todo' : 'done'
        );
      }
      return t;
    }).toList();
  }
}

// Usage in widget
class TaskListScreen extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final tasks = ref.watch(taskListProvider);

    return ListView.builder(
      itemCount: tasks.length,
      itemBuilder: (context, index) => TaskCard(
        title: tasks[index].title,
        status: tasks[index].status,
        onTap: () => ref.read(taskListProvider.notifier).toggleStatus(tasks[index].id),
        onDelete: () => ref.read(taskListProvider.notifier).remove(tasks[index].id),
      ),
    );
  }
}
```

## Model

```dart
// models/task.dart
class Task {
  final String id;
  final String title;
  final String description;
  final String status; // 'todo', 'in_progress', 'done'
  final DateTime createdAt;

  Task({
    required this.id,
    required this.title,
    this.description = '',
    this.status = 'todo',
    DateTime? createdAt,
  }) : createdAt = createdAt ?? DateTime.now();

  Task copyWith({String? title, String? status}) {
    return Task(
      id: id,
      title: title ?? this.title,
      status: status ?? this.status,
      createdAt: createdAt,
    );
  }
}
```

## API Integration

```dart
// services/api_service.dart
import 'dart:convert';
import 'package:http/http.dart' as http;

class ApiService {
  final String baseUrl = 'https://api.example.com';

  Future<List<Task>> getTasks() async {
    final response = await http.get(Uri.parse('$baseUrl/tasks'));
    if (response.statusCode == 200) {
      final List data = jsonDecode(response.body);
      return data.map((json) => Task.fromJson(json)).toList();
    }
    throw Exception('Failed to load tasks');
  }

  Future<Task> createTask(Task task) async {
    final response = await http.post(
      Uri.parse('$baseUrl/tasks'),
      headers: {'Content-Type': 'application/json'},
      body: jsonEncode(task.toJson()),
    );
    return Task.fromJson(jsonDecode(response.body));
  }
}
```

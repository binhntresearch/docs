## Chat Screen — Real-time Messages

```tsx
export default function ChatScreen({ roomId }) {
  const { messages, sendMessage, isTyping, startTyping, stopTyping } = useMessagesStore(roomId);
  const [input, setInput] = useState('');
  const flatListRef = useRef<FlatList>(null);

  const handleSend = () => {
    if (!input.trim()) return;
    sendMessage(input.trim());
    setInput('');
    stopTyping();
  };

  return (
    <View style={styles.container}>
      <FlatList
        ref={flatListRef}
        data={messages}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <MessageBubble message={item} isOwn={item.senderId === currentUserId} />
        )}
        onContentSizeChange={() => flatListRef.current?.scrollToEnd()}
      />

      {isTyping && (
        <Text style={styles.typingIndicator}>Typing...</Text>
      )}

      <View style={styles.inputBar}>
        <TouchableOpacity>
          <Ionicons name="add-circle" size={28} color="#3B82F6" />
        </TouchableOpacity>
        <TextInput
          style={styles.input}
          placeholder="Message..."
          value={input}
          onChangeText={(text) => {
            setInput(text);
            if (text.length > 0) startTyping();
            else stopTyping();
          }}
        />
        <TouchableOpacity onPress={handleSend}>
          <Ionicons name="send" size={28} color="#3B82F6" />
        </TouchableOpacity>
      </View>
    </View>
  );
}
```

## Message Bubble

```tsx
function MessageBubble({ message, isOwn }) {
  return (
    <View style={[styles.bubble, isOwn ? styles.ownBubble : styles.otherBubble]}>
      <Text style={[styles.messageText, isOwn && styles.ownText]}>
        {message.text}
      </Text>
      <View style={styles.messageFooter}>
        <Text style={styles.timestamp}>{formatTime(message.timestamp)}</Text>
        {isOwn && (
          <Ionicons
            name={message.isRead ? "checkmark-done" : "checkmark"}
            size={16}
            color={message.isRead ? "#3B82F6" : "#9CA3AF"}
          />
        )}
      </View>
    </View>
  );
}
```

## WebSocket Chat Service

```dart
// services/chat_service.dart
import 'dart:convert';
import 'package:web_socket_channel/web_socket_channel.dart';

class ChatService {
  WebSocketChannel? _channel;
  final StreamController<Message> _messageController = StreamController.broadcast();

  Stream<Message> get messageStream => _messageController.stream;

  void connect(String userId, String token) {
    _channel = WebSocketChannel.connect(
      Uri.parse('wss://api.example.com/ws?token=$token')
    );

    _channel!.stream.listen((data) {
      final msg = jsonDecode(data);
      if (msg['type'] == 'message') {
        _messageController.add(Message.fromJson(msg['data']));
      } else if (msg['type'] == 'typing') {
        // Handle typing indicator
      } else if (msg['type'] == 'read') {
        // Handle read receipt
      }
    });
  }

  void sendMessage(String roomId, String text) {
    _channel?.sink.add(jsonEncode({
      'type': 'message',
      'roomId': roomId,
      'text': text,
      'timestamp': DateTime.now().toIso8601String()
    }));
  }

  void sendTyping(String roomId, bool isTyping) {
    _channel?.sink.add(jsonEncode({
      'type': 'typing',
      'roomId': roomId,
      'isTyping': isTyping
    }));
  }

  void disconnect() {
    _channel?.sink.close();
    _messageController.close();
  }
}
```

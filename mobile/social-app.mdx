## Home Feed Screen

```tsx
// app/(tabs)/index.tsx
export default function FeedScreen() {
  const { posts, toggleLike, loadMore } = useFeedStore();

  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => item.id}
      renderItem={({ item }) => (
        <PostCard
          post={item}
          onLike={() => toggleLike(item.id)}
          onComment={() => router.push(`/post/${item.id}`)}
          onProfile={() => router.push(`/profile/${item.userId}`)}
        />
      )}
      onEndReached={loadMore}
      onEndReachedThreshold={0.5}
      ListFooterComponent={<ActivityIndicator />}
    />
  );
}

// components/PostCard.tsx
function PostCard({ post, onLike, onComment }) {
  return (
    <View style={styles.post}>
      <TouchableOpacity style={styles.header} onPress={onProfile}>
        <Image source={{ uri: post.avatarUrl }} style={styles.avatar} />
        <Text style={styles.username}>{post.username}</Text>
      </TouchableOpacity>

      <Image source={{ uri: post.images[0] }} style={styles.postImage} />

      <View style={styles.actions}>
        <TouchableOpacity onPress={onLike}>
          <Ionicons
            name={post.isLikedByMe ? "heart" : "heart-outline"}
            size={28}
            color={post.isLikedByMe ? "#EF4444" : "#000"}
          />
        </TouchableOpacity>
        <TouchableOpacity onPress={onComment}>
          <Ionicons name="chatbubble-outline" size={26} />
        </TouchableOpacity>
        <TouchableOpacity>
          <Ionicons name="paper-plane-outline" size={26} />
        </TouchableOpacity>
      </View>

      <Text style={styles.likes}>{post.likesCount} likes</Text>
      <Text style={styles.caption}>
        <Text style={styles.username}>{post.username}</Text> {post.content}
      </Text>
    </View>
  );
}
```

## Create Post

```tsx
export default function CreatePostScreen() {
  const [images, setImages] = useState<string[]>([]);
  const [caption, setCaption] = useState('');
  const { createPost, isUploading } = usePostStore();
  const router = useRouter();

  const pickImages = async () => {
    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsMultipleSelection: true,
      quality: 0.8
    });

    if (!result.canceled) {
      setImages(result.assets.map(a => a.uri));
    }
  };

  const handlePost = async () => {
    await createPost({ images, caption });
    router.back();
  };

  return (
    <View style={styles.container}>
      <FlatList
        horizontal
        data={images}
        renderItem={({ item }) => <Image source={{ uri: item }} style={styles.preview} />}
        ListEmptyComponent={
          <TouchableOpacity style={styles.addImage} onPress={pickImages}>
            <Ionicons name="images" size={48} color="#9CA3AF" />
          </TouchableOpacity>
        }
      />
      <TextInput
        style={styles.captionInput}
        placeholder="Write a caption..."
        multiline
        value={caption}
        onChangeText={setCaption}
      />
      <TouchableOpacity style={styles.postBtn} onPress={handlePost}>
        <Text style={styles.postBtnText}>Share</Text>
      </TouchableOpacity>
    </View>
  );
}
```

Backend servers xử lý **toàn bộ logic nghiệp vụ** — bảo mật server, database, network là layer quan trọng nhất.

## Attack Surfaces của Backend

```
┌─────────────────────────────────────┐
│      Internet (Public)               │
└──────────────┬──────────────────────┘
               ↓ HTTPS
┌──────────────▼──────────────────────┐
│      Load Balancer / WAF            │ ← Rate limiting, DDoS protection
└──────────────┬──────────────────────┘
               ↓
┌──────────────▼──────────────────────┐
│      Application Server              │ ← TLS, authentication
│  (Express, FastAPI, Spring Boot)    │
└──────────────┬──────────────────────┘
               ↓ Internal Network
┌──────────────▼──────────────────────┐
│      Database Server                │ ← Encrypted at rest
│  (PostgreSQL, MySQL, MongoDB)       │
└─────────────────────────────────────┘
```

## 1. Server Hardening

### HTTPS Only

```nginx
# Nginx config - force HTTPS
server {
    listen 80;
    server_name example.com;

    # Redirect HTTP to HTTPS
    location / {
        return 301 https://$server_name$request_uri;
    }
}

server {
    listen 443 ssl http2;
    server_name example.com;

    ssl_certificate /etc/letsencrypt/live/example.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/example.com/privkey.pem;

    # Modern TLS config
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers 'ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384';
    ssl_prefer_server_ciphers on;

    # TLS 1.3 specific
    ssl_early_data on;  # 0-RTT (use carefully)

    # Security headers
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;
    add_header X-Content-Type-Options nosniff always;
    add_header X-Frame-Options DENY always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
}
```

### Encryption at Rest

```python
from cryptography.fernet import Fernet
from sqlalchemy.types import TypeDecorator, LargeBinary

class EncryptedString(TypeDecorator):
    """Encrypted string column type"""

    impl = LargeBinary
    cache_ok = True

    def __init__(self, key: bytes):
        super().__init__()
        self.cipher = Fernet(key)

    def process_bind_param(self, value, dialect):
        if value is None:
            return None
        return self.cipher.encrypt(value.encode())

    def process_result_value(self, value, dialect):
        if value is None:
            return None
        return self.cipher.decrypt(bytes(value)).decode()

# Usage
ENCRYPTION_KEY = Fernet.generate_key()  # Store securely!

class Customer(Base):
    __tablename__ = "customers"

    id = Column(Integer, primary_key=True)
    ssn = Column(EncryptedString(ENCRYPTION_KEY))  # Encrypted!
    email = Column(String, unique=True)
```

## 3. Authentication Implementation

```python
from fastapi import Depends, HTTPException, status, Header
from passlib.context import CryptContext

pwd_context = CryptContext(
    schemes=["bcrypt"],
    deprecated="auto",
    bcrypt__rounds=12
)

def hash_password(plain: str) -> str:
    return pwd_context.hash(plain)

def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)

# ❌ Common mistakes
async def authenticate(username, password):
    user = db.users.find_by(username=username)
    # ❌ Timing attack - find_by or verify sequentially
    if not user or not verify_password(password, user.password_hash):
        raise HTTPException(401)
    return user

# ✅ CORRECT - constant-time comparison
async def authenticate(username, password):
    user = db.users.find_by(username=username)

    # Always run hash check (even with fake hash)
    fake_hash = "$2b$12$" + "0" * 53  # Constant bcrypt hash
    target_hash = user.password_hash if user else fake_hash

    is_valid = pwd_context.verify(password, target_hash)

    if not user or not is_valid:
        raise HTTPException(401, "Invalid credentials")

    return user
```

## 4. Session Security

```python
from fastapi import Response
import secrets

@app.post("/login")
async def login(response: Response, request: LoginRequest):
    user = await authenticate(request.username, request.password)
    if not user:
        raise HTTPException(401)

    # Generate cryptographically secure session ID
    session_id = secrets.token_urlsafe(32)

    # Store in Redis with TTL
    redis_client.setex(
        f"session:{session_id}",
        3600,  # 1 hour
        json.dumps({"user_id": user.id, "ip": request.client.host})
    )

    # Set secure HTTPOnly cookie
    response.set_cookie(
        key="session_id",
        value=session_id,
        max_age=3600,
        httponly=True,           # Cannot be accessed by JS
        secure=True,             # Only over HTTPS
        samesite="strict",       # CSRF protection
        path="/"
    )
    return {"message": "Logged in"}
```

## 6. Error Handling — Don't Leak Info

```python
from fastapi import Request
from fastapi.responses import JSONResponse

@app.exception_handler(Exception)
async def exception_handler(request: Request, exc: Exception):
    # Log full error internally
    logger.error(f"Unhandled error: {exc}", exc_info=True, extra={
        "path": request.url.path,
        "method": request.method
    })

    # Return generic error to client
    if settings.debug:
        return JSONResponse(
            status_code=500,
            content={"error": str(exc), "type": type(exc).__name__}
        )

    # ❌ NEVER - leaks internal info
    # return {"error": str(exc), "traceback": traceback.format_exc()}

    # ✅ SAFE
    return JSONResponse(
        status_code=500,
        content={"error": "Internal server error", "request_id": request_id}
    )
```

## 7. Audit Logging

```python
from sqlalchemy import event
from sqlalchemy.orm import Session

def setup_audit_logging():
    """Log ALL changes to sensitive tables"""

    @event.listens_for(User, "after_update")
    @event.listens_for(User, "after_insert")
    @event.listens_for(User, "after_delete")
    def log_user_change(mapper, connection, target):
        logger.info("user_change", extra={
            "table": "users",
            "user_id": target.id,
            "action": "modified",
            "changed_by": current_user_id(),  # Thread-local
            "ip": current_request_ip()
        })

setup_audit_logging()
```

## 9. Rate Limiting at Server Level

```nginx
# Nginx rate limiting
http {
    limit_req_zone $binary_remote_addr zone=login:10m rate=5r/m;

    server {
        location /auth/login {
            limit_req zone=login burst=5 nodelay;
            proxy_pass http://backend;
        }

        location /api/ {
            limit_req zone=api burst=20 nodelay;
            proxy_pass http://backend;
        }
    }

    limit_req_zone $binary_remote_addr zone=api:10m rate=100r/m;
}

# DDoS protection
limit_conn_zone $binary_remote_addr zone=conn_per_ip:10m;
limit_conn conn_per_ip 50;
```

## 10. Container Security

```dockerfile
# Dockerfile - production hardening
FROM python:3.11-slim

# Run as non-root user
RUN useradd -m -u 1000 appuser

# Use multi-stage build to reduce attack surface
COPY --from=builder /app /app

# Security: read-only filesystem
USER appuser
WORKDIR /app

# Drop ALL capabilities
RUN setcap --remove-all 'cap_net_bind_service' /usr/bin/python3

# Health check
HEALTHCHECK --interval=30s --timeout=3s CMD curl -f http://localhost:8000/health || exit 1

CMD ["gunicorn", "--bind", "0.0.0.0:8000", "app:app"]
```

```yaml
# docker-compose.yml - security hardening
services:
  api:
    image: my-api:latest
    read_only: true                 # Read-only filesystem

    security_opt:
      - no-new-privileges:true     # Block privilege escalation
      - apparmor:docker-default
    cap_drop:
      - ALL                         # Drop all Linux capabilities
    cap_add:
      - NET_BIND_SERVICE
    tmpfs:
      - /tmp                        # Writable temp dir
```

## API Gateway

```typescript
// api-gateway/src/server.ts
import express from 'express';
import httpProxy from 'http-proxy-middleware';

const app = express();

const SERVICES = {
  users: 'http://user-service:3001',
  tasks: 'http://task-service:3002',
  notifications: 'http://notification-service:3003'
};

// Verify JWT at gateway
const verifyJWT = (req, res, next) => {
  const token = req.headers.authorization?.replace('Bearer ', '');
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    res.status(401).json({ error: 'Invalid token' });
  }
};

// Proxy routes
app.use('/api/users', verifyJWT, httpProxy(SERVICES.users));
app.use('/api/tasks', verifyJWT, httpProxy(SERVICES.tasks));
app.use('/api/notifications', verifyJWT, httpProxy(SERVICES.notifications));

app.listen(3000, () => console.log('Gateway on port 3000'));
```

## Task Service

```typescript
// task-service với RabbitMQ event publishing
import amqp from 'amqplib';

class TaskService {
  async createTask(userId: number, taskData: any) {
    const task = await db.tasks.insert({ ...taskData, owner_id: userId });

    // Publish event
    await this.publishEvent('task.created', {
      taskId: task.id,
      ownerId: userId,
      task
    });

    return task;
  }

  private async publishEvent(eventName: string, data: any) {
    const connection = await amqp.connect(process.env.RABBITMQ_URL);
    const channel = await connection.createChannel();
    await channel.assertExchange('tasks', 'topic');
    channel.publish('tasks', eventName, Buffer.from(JSON.stringify(data)));
    await channel.close();
    await connection.close();
  }
}
```

## Notification Service

```typescript
// notification-service consumes events
import amqp from 'amqplib';

class NotificationService {
  async start() {
    const connection = await amqp.connect(process.env.RABBITMQ_URL);
    const channel = await connection.createChannel();

    await channel.assertExchange('tasks', 'topic');
    const queue = await channel.assertQueue('notifications');
    await channel.bindQueue(queue.queue, 'tasks', 'task.#');

    channel.consume(queue.queue, async (msg) => {
      const event = JSON.parse(msg.content.toString());
      const routingKey = msg.fields.routingKey;

      if (routingKey === 'task.created') {
        await this.sendEmail(event.ownerId, 'New task created', event.task);
      }
      if (routingKey === 'task.completed') {
        await this.sendEmail(event.ownerId, 'Task completed!', event.task);
      }

      channel.ack(msg);
    });
  }

  async sendEmail(userId: number, subject: string, task: any) {
    const user = await fetch(`http://user-service:3001/users/${userId}`).then(r => r.json());
    await emailService.send(user.email, subject, task);
  }
}
```

## Service Discovery

```typescript
// Simple service registry
const SERVICE_REGISTRY = {
  'user-service': ['http://user-service-1:3001', 'http://user-service-2:3001'],
  'task-service': ['http://task-service-1:3002']
};

class ServiceRegistry {
  static getInstance(serviceName: string): string {
    const instances = SERVICE_REGISTRY[serviceName];
    return instances[Math.floor(Math.random() * instances.length)];
  }
}

// Load balancing
const userUrl = ServiceRegistry.getInstance('user-service');
const response = await fetch(`${userUrl}/users/${userId}`);
```

## Saga Pattern cho Distributed Transactions

```typescript
// Distributed transaction cho create order (task + payment)
class OrderSaga {
  async execute(orderData: any) {
    try {
      // Step 1: Create task
      const task = await taskService.createTask(orderData);

      // Step 2: Process payment
      const payment = await paymentService.charge(orderData.payment);

      // Step 3: Confirm task
      await taskService.confirmTask(task.id, payment.id);

      return { task, payment };
    } catch (error) {
      // Compensating actions
      if (error.step === 1) throw error;
      if (error.step === 2) await taskService.cancelTask(orderData.taskId);
      throw error;
    }
  }
}
```

## Best Practices

<Tip>
- **Database per service:** Microservices có database riêng
- **API contracts:** OpenAPI spec cho mỗi service
- **Async communication:** Message queue cho cross-service events
- **Circuit breaker:** Fail fast khi service down
- **Distributed tracing:** Track request across services (Jaeger)
- **Container orchestration:** Use Kubernetes từ 5+ services
</Tip>

Testing là critical — bảo đảm code hoạt động đúng và refactor an toàn. Có 3 levels: **Unit → Integration → E2E**.

## Testing Pyramid

```
        /\
       /E2E\         ← Slow, expensive, few tests
      /------\
     /Integration\    ← Medium speed, moderate coverage
    /----------\
   /   Unit Tests   \  ← Fast, cheap, many tests
  /------------------\
```

## Unit Tests — Node.js (Jest)

```typescript
// services/UserService.test.ts
import { UserService } from './UserService';

describe('UserService', () => {
  let service: UserService;
  let mockDb: any;

  beforeEach(() => {
    mockDb = {
      users: {
        create: jest.fn(),
        findByEmail: jest.fn(),
        findById: jest.fn()
      }
    };
    service = new UserService(mockDb);
  });

  describe('createUser', () => {
    it('should create user with hashed password', async () => {
      mockDb.users.findByEmail.mockResolvedValue(null);
      mockDb.users.create.mockResolvedValue({ id: 1, email: 'alice@x.com' });

      const user = await service.create({
        email: 'alice@x.com',
        password: 'plainpass'
      });

      expect(user.id).toBe(1);
      expect(mockDb.users.create).toHaveBeenCalledWith(
        expect.objectContaining({
          email: 'alice@x.com',
          password: expect.not.stringMatching('plainpass')
        })
      );
    });

    it('should throw if email exists', async () => {
      mockDb.users.findByEmail.mockResolvedValue({ id: 1, email: 'a@x.com' });

      await expect(service.create({ email: 'a@x.com', password: 'p' }))
        .rejects.toThrow('Email already exists');
    });
  });
});
```

## Unit Tests — Python (pytest)

```python
import pytest
from services.user_service import UserService

@pytest.fixture
def mock_db():
    return MagicMock()

@pytest.fixture
def service(mock_db):
    return UserService(db=mock_db)

class TestUserService:
    async def test_create_user_hashes_password(self, service, mock_db):
        mock_db.users.find_by_email.return_value = None
        mock_db.users.create.return_value = {"id": 1, "email": "alice@x.com"}

        user = await service.create(email="alice@x.com", password="plainpass")

        assert user["id"] == 1
        assert mock_db.users.create.called_with(
            mock_db.users.create.call_args[0][0].password != "plainpass"
        )

    async def test_create_user_duplicate_email(self, service, mock_db):
        mock_db.users.find_by_email.return_value = {"id": 1}

        with pytest.raises(ValueError, match="Email already exists"):
            await service.create(email="a@x.com", password="p")
```

## Integration Tests

```python
# tests/integration/test_users.py
import pytest
from httpx import AsyncClient

@pytest.mark.asyncio
async def test_create_and_login(client: AsyncClient):
    # Create user
    response = await client.post("/api/users", json={
        "email": "test@example.com",
        "password": "securepass123",
        "name": "Test User"
    })
    assert response.status_code == 201
    user_id = response.json()["id"]

    # Login
    response = await client.post("/auth/login", json={
        "email": "test@example.com",
        "password": "securepass123"
    })
    assert response.status_code == 200
    token = response.json()["access_token"]

    # Get profile (protected)
    response = await client.get("/me", headers={"Authorization": f"Bearer {token}"})
    assert response.status_code == 200
    assert response.json()["email"] == "test@example.com"
```

## Mocking External Services

```python
# Mock email service
@pytest.fixture
def mock_email_service(monkeypatch):
    sent_emails = []

    async def fake_send(to, subject, body):
        sent_emails.append({"to": to, "subject": subject})

    monkeypatch.setattr("services.email.send", fake_send)
    return sent_emails

async def test_signup_sends_email(mock_email_service):
    await client.post("/api/users", json={...})
    assert len(mock_email_service) == 1
    assert mock_email_service[0]["to"] == "user@x.com"
```

## Coverage

```bash
# Run tests with coverage
pytest --cov=app --cov-report=html
open htmlcov/index.html

# Aim 80%+ coverage, 100% cho critical paths (auth, payments)
```

<Tip>
Test strategy cho Claude Code:
- **Start với failing test** (TDD) hoặc viết test đầu tiên
- **Mock everything external** — database, HTTP, services
- **Test edge cases:** null, empty, max length, special chars
- **Test error paths:** invalid auth, not found, validation errors
- **Maintain test coverage:** Watch for drops in CI
</Tip>
